* 同一个页面存在两个控制器：

<div data-ng-controller="jackController">
</div>
....
<div data-ng-controller="demoController">
</div>


* jackController.js 内容：

	var appElement = document.querySelector('[data-ng-controller=demoController]');
	var scope = angular.element(appElement).scope();
	scope.load();	// 这里可以调用成功！


* demoController.js：
	$scope.load = function(){
		console.log('我被执行了');
	}
